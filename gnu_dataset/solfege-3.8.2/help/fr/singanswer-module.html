The singanswer module

Here is a minimal lesson file:

header {
  module = singanswer
  lesson_id = "a400df62-e007-4a1b-9057-cd05397e88a2"
  version = "3.1.4"
  title = "Sing the root of the chord"
}
question {
    question_text = "Sing the root"
    music = chord("c' e' g'")
    answer = chord("c'")
}
question {
    question_text = "Sing the root"
    music = chord("a' c'' e''")
    answer = chord("a'")
}
